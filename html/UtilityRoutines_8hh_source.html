275  template <typename A> inline int FindItemInSortedList(std::string const &String, MArray1<A, std::string> const &ListOfItems, int const NumItems)
276  {
277  int Probe(0);
278  int LBnd(0);
279  int UBnd(NumItems + 1);
280  bool Found(false);
281  while ((!Found) || (Probe != 0)) {
282  Probe = (UBnd - LBnd) / 2;
283  if (Probe == 0) break;
284  Probe += LBnd;
285  if (equali(String, ListOfItems(Probe))) {
286  Found = true;
287  break;
288  } else if (lessthani(String, ListOfItems(Probe))) {
289  UBnd = Probe;
290  } else {
291  LBnd = Probe;
292  }
293  }
294  return Probe;
295  }
