232  // Line Segment Intersects Enclosing Sphere?
233  bool segmentIntersectsSphere(Vertex const &a, Vertex const &b) const
234  {
235  Vertex const ab(b - a);
236  Real const ab_mag_squared(ab.mag_squared());
237  if (ab_mag_squared == 0.0) { // Segment is a point
238  return ObjexxFCL::distance_squared(a, c_) <= r_;
239  } else { // Might pay to check if a or b in sphere first in some applications
240  Vertex const ac(c_ - a);
241  Real const projection_fac(((ac.x * ab.x) + (ac.y * ab.y) + (ac.z * ab.z)) / ab_mag_squared);
242  if ((0.0 <= projection_fac) && (projection_fac <= 1.0)) { // Projected (closest) point is on ab segment
243  return ObjexxFCL::distance_squared(ac, projection_fac * ab) <= r_;
244  } else { // Projection (closest) point is outside of ab segment: Intersects iff a or b are in sphere
245  return (ObjexxFCL::distance_squared(a, c_) <= r_) || (ObjexxFCL::distance_squared(b, c_) <= r_);
246  }
247  }
248  }

250  // Ray Intersects Enclosing Sphere?
251  bool rayIntersectsSphere(Vertex const &a, Vertex const &dir) const
252  {
253  assert(std::abs(dir.mag_squared() - 1.0) < 4 * std::numeric_limits<Real>::epsilon()); // Check unit vector
254  // Might pay to check if a in sphere first in some applications
255  Vertex const ac(c_ - a);
256  Real const projection_fac((ac.x * dir.x) + (ac.y * dir.y) + (ac.z * dir.z));
257  if (0.0 <= projection_fac) { // Projected (closest) point is on ray
258  return ObjexxFCL::distance_squared(ac, projection_fac * dir) <= r_;
259  } else { // Projection (closest) point is outside of ray: Intersects iff a is in sphere
260  return ObjexxFCL::distance_squared(a, c_) <= r_;
261  }

272  // Line Segment Intersects Cube?
273  bool segmentIntersectsCube(Vertex const &a, Vertex const &b) const
274  {
275  // Check if either segment endpoint is in cube
276  if (contains(a) || contains(b)) return true;
277 
278  // Use separating axis theorem (faster variants exist)
279  Vertex const m(mid(a, b) - c_); // Mid-point relative to cube center
280  Vertex const mb(b - c_ - m); // ab mid-point to b half segment vector
281  Vertex const e(std::abs(mb.x), std::abs(mb.y), std::abs(mb.z)); // Extent of half ab segment
282  Real const h(0.5 * w_); // Half-width
283  // Check if x,y,z axes are separating
284  if (std::abs(m.x) > h + e.x) return false;
285  if (std::abs(m.y) > h + e.y) return false;
286  if (std::abs(m.z) > h + e.z) return false;
287  // Check if cross product axes are separating
288  if (std::abs(m.y * mb.z - m.z * mb.y) > h * (e.z + e.y)) return false;
289  if (std::abs(m.x * mb.z - m.z * mb.x) > h * (e.z + e.x)) return false;
290  if (std::abs(m.x * mb.y - m.y * mb.x) > h * (e.y + e.x)) return false;
291  return true;
292  }

294  // Ray Intersects Cube?
295  bool rayIntersectsCube(Vertex const &a, Vertex const &dir, Vertex const &dir_inv) const
296  {
297  // Note: dir_inv coordinates corresponding to a zero dir coordinate are not used and can be set to zero
298 
299  assert(std::abs(dir.mag_squared() - 1.0) < 4 * std::numeric_limits<Real>::epsilon()); // Check unit vector
300  assert((dir.x == 0.0) || (std::abs(dir_inv.x - (1.0 / dir.x)) < 2 * std::numeric_limits<Real>::epsilon() * std::abs(dir_inv.x)));
301  assert((dir.y == 0.0) || (std::abs(dir_inv.y - (1.0 / dir.y)) < 2 * std::numeric_limits<Real>::epsilon() * std::abs(dir_inv.y)));
302  assert((dir.z == 0.0) || (std::abs(dir_inv.z - (1.0 / dir.z)) < 2 * std::numeric_limits<Real>::epsilon() * std::abs(dir_inv.z)));
303 
304  // Check if ray origin is in cube
305  if (contains(a)) return true;
306 
307  // Refined Smits' method: Largest distance to 3 visible cube face planes along ray is the candidate entry point
308  Real tx, ty, tz; // Ray position parameter for intersections with box face planes
309  if (dir.x > 0.0) {
310  tx = (l_.x - a.x) * dir_inv.x;
311  } else if (dir.x < 0.0) {
312  tx = (u_.x - a.x) * dir_inv.x;
313  } else { // dir.x == 0
314  tx = std::numeric_limits<Real>::lowest();
315  }
316  if (dir.y > 0.0) {
317  ty = (l_.y - a.y) * dir_inv.y;
318  } else if (dir.y < 0.0) {
319  ty = (u_.y - a.y) * dir_inv.y;
320  } else { // dir.y == 0
321  ty = std::numeric_limits<Real>::lowest();
322  }
323  if (dir.z > 0.0) {
324  tz = (l_.z - a.z) * dir_inv.z;
325  } else if (dir.z < 0.0) {
326  tz = (u_.z - a.z) * dir_inv.z;
327  } else { // dir.z == 0
328  tz = std::numeric_limits<Real>::lowest();
329  }
330 
331  Real const tmax(ObjexxFCL::max(tx, ty, tz));
332  if (tmax >= 0.0) { // Entry point is on ray: See if it is within cube extent
333  if (tx == tmax) {
334  Real const y(a.y + (tmax * dir.y));
335  if ((y < l_.y) || (y > u_.y)) return false;
336  Real const z(a.z + (tmax * dir.z));
337  if ((z < l_.z) || (z > u_.z)) return false;
338  } else if (ty == tmax) {
339  Real const x(a.x + (tmax * dir.x));
340  if ((x < l_.x) || (x > u_.x)) return false;
341  Real const z(a.z + (tmax * dir.z));
342  if ((z < l_.z) || (z > u_.z)) return false;
343  } else { // tz == tmax
344  Real const x(a.x + (tmax * dir.x));
345  if ((x < l_.x) || (x > u_.x)) return false;
346  Real const y(a.y + (tmax * dir.y));
347  if ((y < l_.y) || (y > u_.y)) return false;
348  }
349  return true;
350  } else { // Entry point is on backwards projection of ray
351  return false;
352  }
353  }

361  // Line Intersects Cube?
362  bool lineIntersectsCube(Vertex const &a, Vertex const &dir, Vertex const &dir_inv) const
363  {
364  // Note: dir_inv coordinates corresponding to a zero dir coordinate are not used and can be set to zero
365 
366  assert(std::abs(dir.mag_squared() - 1.0) < 4 * std::numeric_limits<Real>::epsilon()); // Check unit vector
367  assert((dir.x == 0.0) || (std::abs(dir_inv.x - (1.0 / dir.x)) < 2 * std::numeric_limits<Real>::epsilon() * std::abs(dir_inv.x)));
368  assert((dir.y == 0.0) || (std::abs(dir_inv.y - (1.0 / dir.y)) < 2 * std::numeric_limits<Real>::epsilon() * std::abs(dir_inv.y)));
369  assert((dir.z == 0.0) || (std::abs(dir_inv.z - (1.0 / dir.z)) < 2 * std::numeric_limits<Real>::epsilon() * std::abs(dir_inv.z)));
370 
371  // Smits' method: Largest distance to 3 visible cube faces along ray is the candidate entry point
372  Real txl, txu, tyl, tyu, tzl, tzu; // Ray position parameter for intersections with box face planes
373  if (dir.x > 0.0) {
374  txl = (l_.x - a.x) * dir_inv.x;
375  txu = (u_.x - a.x) * dir_inv.x;
376  } else if (dir.x < 0.0) {
377  txl = (u_.x - a.x) * dir_inv.x;
378  txu = (l_.x - a.x) * dir_inv.x;
379  } else { // dir.x == 0
380  if ((l_.x - a.x <= 0.0) && (u_.x - a.x >= 0.0)) {
381  txl = std::numeric_limits<Real>::lowest();
382  txu = std::numeric_limits<Real>::max();
383  } else {
384  return false;
385  }
386  }
387  if (dir.y > 0.0) {
388  tyl = (l_.y - a.y) * dir_inv.y;
389  tyu = (u_.y - a.y) * dir_inv.y;
390  } else if (dir.y < 0.0) {
391  tyl = (u_.y - a.y) * dir_inv.y;
392  tyu = (l_.y - a.y) * dir_inv.y;
393  } else { // dir.y == 0
394  if ((l_.y - a.y <= 0.0) && (u_.y - a.y >= 0.0)) {
395  tyl = std::numeric_limits<Real>::lowest();
396  tyu = std::numeric_limits<Real>::max();
397  } else {
398  return false;
399  }
400  }
401  if ((txl > tyu) || (tyl > txu)) return false;
402  if (dir.z > 0.0) {
403  tzl = (l_.z - a.z) * dir_inv.z;
404  tzu = (u_.z - a.z) * dir_inv.z;
405  } else if (dir.z < 0.0) {
406  tzl = (u_.z - a.z) * dir_inv.z;
407  tzu = (l_.z - a.z) * dir_inv.z;
408  } else { // dir.z == 0
409  if ((l_.z - a.z <= 0.0) && (u_.z - a.z >= 0.0)) {
410  tzl = std::numeric_limits<Real>::lowest();
411  tzu = std::numeric_limits<Real>::max();
412  } else {
413  return false;
414  }
415  }
416  return ((txl <= tzu) && (tzl <= txu) && (tyl <= tzu) && (tzl <= tyu));
417  }
