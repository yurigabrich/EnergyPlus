74  class pvwatts_celltemp
75  {
76  /*
77  This class was converted from the original PVWatts subroutine
78  to store the previous irradiance and cell temp to make the model
79  give consistent results with the online PVWatts V1. Previously
80  all calculations were done over the course of a day, so the
81  previous module/sun values were saved, but in the single time stamp
82  version, these changes were not tracked. apd 2/24/2012
83 
84  Defines function to calculate cell temperature, changed 8/22/2007 to
85  work with single time stamp data, also see pvsubs2.c
86 
87  This function was converted from a PVFORM version 3.3 subroutine
88  c this routine estimates the array temperature given the poa radiation,
89  c ambient temperature, and wind speed. it uses an advanced cell temp
90  c model developed by m fuentes at snla. if the poa insolation is eq
91  c zero then set cell temp = 999.
92  c
93  passed variables:
94  inoct = installed nominal operating cell temperature (deg K)
95  height = average array height (meters)
96  poa2 = plane of array irradiances (W/m2)
97  ws2 = wind speeds (m/s)
98  ambt2 = ambient temperatures (deg C)
99 
100  c local variables :
101  c absorb = absorbtivity
102  c backrt = ratio of actual backside heat xfer to theoretical of rack mount
103  c boltz = boltzmann's constant
104  c cap = capacitance per unit area of module
105  c capo = capacitance per unit area of rack mounted module
106  c conair = conductivity of air
107  c convrt = ratio of total convective heat xfer coef to topside hxc
108  c denair = density of air
109  c dtime = time step
110  c eigen = product of eigen value and time step
111  c emmis = emmisivity
112  ex = ?
113  c grashf = grashoffs number
114  c hconv = convective coeff of module (both sides)
115  c hforce = forced convective coeff of top side
116  c hfree = free convective coeff of top side
117  c hgrnd = radiative heat xfer coeff from module to ground
118  hsky = ?
119  c iflagc = flag to check if routine has been executed
120  c reynld = reynolds number
121  c suun = insolation at start of time step
122  c suno = previous hours insolation
123  c tamb = ambient temp
124  c tave = average of amb and cell temp
125  c tgrat = ratio of grnd temp above amb to cell temp above amb
126  c tgrnd = temperature of ground
127  c tmod = computed cell temp
128  c tmodo = cell temp for previous time step
129  c tsky = sky temp
130  c visair = viscosity of air
131  c windmd = wind speed at module height
132  c xlen = hydrodynamic length of module */
133

138  double eigen, emmis, grashf, hconv, hforce, hfree, hgrnd, reynld, suun;
139  double suno, tamb, tave, tgrat, tgrnd, tmod, tmodo, tsky, visair, windmd, xlen;
