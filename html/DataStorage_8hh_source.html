78  template <typename T> std::vector<T *> addObjects(json const &objs)
79  {
80  assert(objs.is_object());

131  template <typename T> auto unique_void(T *ptr) -> unique_void_ptr
132  {
133  auto deleter = [](void *data) -> void {
