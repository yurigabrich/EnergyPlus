51 // Purpose: Functions for checking if a ray hits a surface
52 //
53 // Author: Stuart Mentzer (Stuart_Mentzer@objexx.com)
54 //
55 // History:
56 // Jun 2015: Last update of legacy version based on DOE-2 DPIERC
57 // Jan 2016: Initial release
58 //
59 // Notes:
60 // This is filling the role of the former PierceSurface function authored by Fred Winkelmann and based on
61 // DOE-2.1E subroutine DPIERC and some aspects of this version are analogous
62 // To match the former behavior rays with origin exactly on the surface are treated as not hitting
63 // These functions are VERY performance critical for daylighting and solar reflection
64 // This high-performance implementation was built to complement the octree system for scalability of those systems
65 // This has been carefully designed for speed but is probably not be optimal yet
66 // For EnergyPlus most surfaces are rectangular so that is the most important for performance
67 // Inlining, storing preprocessed values in Surface, 2D projection, & short circuiting are used here for speed
68 // Agressive inlining options may be needed to get peak performance
69 // Don't make changes here without validating the performance impact

112 inline void PierceSurface_Convex(DataSurfaces::Surface2D const &s2d, // 2D surface
113  Vector2<Real64> const &h2d, // 2D hit point
114  bool &hit // Ray intersects surface?
115 )
116 {
117  // Purpose: Check if a 2D hit point is in a convex 2D surface
118  //
119  // Author: Stuart Mentzer (Stuart_Mentzer@objexx.com)
120  //
121  // History:
122  // Jan 2016: Initial release
123  //
124  // Notes:
125  // Pulled this rare case out into separate function to facilitate inlining
126  // This is O( n ) complexity so it is isn't used for many-vertex surfaces
127 
128  using DataSurfaces::Surface2D;
129  Surface2D::Vertices const &vs(s2d.vertices); // 2D surface vertices
130  Surface2D::Vectors const &es(s2d.edges); // 2D surface edge vectors
131  Surface2D::Vertices::size_type const n(vs.size());
132  assert(n >= 3u);
133  switch (n) {
134  case 8:
135  if (es[7].cross(h2d - vs[7]) < 0.0) {
136  return;
137  }
138  // fallthrough
139  case 7:
140  if (es[6].cross(h2d - vs[6]) < 0.0) {
141  return;
142  }
143  // fallthrough
144  case 6:
145  if (es[5].cross(h2d - vs[5]) < 0.0) {
146  return;
147  }
148  // fallthrough
149  case 5:
150  if (es[4].cross(h2d - vs[4]) < 0.0) {
151  return;
152  }
153  // fallthrough
154  case 4:
155  if (es[3].cross(h2d - vs[3]) < 0.0) {
156  return;
157  }
158  // fallthrough
159  case 3:
160  if (es[2].cross(h2d - vs[2]) < 0.0) {
161  return;
162  }
163  if (es[1].cross(h2d - vs[1]) < 0.0) {
164  return;
165  }
166  if (es[0].cross(h2d - vs[0]) < 0.0) {
167  return;
168  }
169  hit = true;
170  return;
171  default:
172  for (Surface2D::Vertices::size_type i = 0; i < n; ++i) {
173  if (es[i].cross(h2d - vs[i]) < 0.0) return;
174  }
175  hit = true;
176  return;
177  }
178 }

180 inline void PierceSurface_Nonconvex(DataSurfaces::Surface2D const &s2d, // 2D surface
181  Vector2<Real64> const &h2d, // 2D hit point
182  bool &hit // Ray intersects surface?
183 )
184 {
185  // Purpose: Check if a 2D hit point is in a 2D possibly nonconvex surface
186  //
187  // Author: Stuart Mentzer (Stuart_Mentzer@objexx.com)
188  //
189  // History:
190  // Jan 2016: Initial release
191  //
192  // Notes:
193  // Pulled this rare case out into separate function to facilitate inlining
194  // This works for nonconvex "simple" (no edge crossings) polygons
195  // This is also a fast O( log n ) algorithm for many-vertex convex surfaces
196 
197  using DataSurfaces::Surface2D;
198  using size_type = Surface2D::Vertices::size_type;
199  using Slab = DataSurfaces::Surface2DSlab;
200  using Vertex2D = Vector2<Real64>;
201  assert(s2d.vertices.size() >= 3u);
202  Surface2D::Slabs const &slabs(s2d.slabs); // 2D surface y slice slabs
203  Surface2D::SlabYs const &slabYs(s2d.slabYs); // 2D surface slab y coordinates
204  assert(slabYs.size() > 0u);
205  Real64 const yHit(h2d.y); // Hit point y coordinate
206 
207  // Find slab with y range containing hit point
208  auto const iHit(std::lower_bound(slabYs.begin(), slabYs.end(), yHit));
209  assert((yHit >= slabYs.front()) && (yHit <= slabYs.back())); // Passed bounding box check so hit point in slabs y range
210  assert(iHit != slabYs.end()); // Hit point can't be above all slabs: passed bounding box check
211  size_type const iSlab(std::min(static_cast<size_type>(iHit - 1 - slabYs.begin()), slabs.size())); // Hit slab index
212  Slab const &slab(slabs[iSlab]);
213 
214  // Check hit point within slab bounding box x range
215  Real64 const xHit(h2d.x); // Hit point x coordinate
216  if ((xHit < slab.xl) || (xHit > slab.xu)) return; // Hit point outside slab bounding box
217 
218  // Find edge pair surrounding hit point
219  Slab::Edges const &slabEdges(slab.edges);
220  Slab::EdgesXY const &slabEdgesXY(slab.edgesXY);
221  size_type const nEdges(slabEdges.size());
222  assert(nEdges >= 2u);
223  if (nEdges == 2) { // 2 edges
224  Slab::Edge const se0(slabEdges[0]);
225  Slab::EdgeXY const eXY0(slabEdgesXY[0]);
226  Vertex2D v0(s2d.vertices[se0]);
227  Surface2D::Edge e0(s2d.edges[se0]);
228  Real64 const x0(v0.x + (yHit - v0.y) * eXY0);
229  if (xHit < x0) return; // Hit point x is left of left edge
230  Slab::Edge const se1(slabEdges[1]);
231  Slab::EdgeXY const eXY1(slabEdgesXY[1]);
232  Vertex2D v1(s2d.vertices[se1]);
233  Surface2D::Edge e1(s2d.edges[se1]);
234  Real64 const x1(v1.x + (yHit - v1.y) * eXY1);
235  if (x1 < xHit) return; // Hit point is right of right edge
236  } else { // 4+ edges: Binary search for edges surrounding hit point
237  assert(nEdges >= 4u);
238  assert(nEdges % 2 == 0u);
239  size_type l(0u), u(nEdges - 1);
240  Slab::Edge const il(slabEdges[l]);
241  Slab::EdgeXY const eXYl(slabEdgesXY[l]);
242  Vertex2D const &vl(s2d.vertices[il]);
243  Surface2D::Edge const el(s2d.edges[il]);
244  Real64 const xl(vl.x + (yHit - vl.y) * eXYl);
245  if (xHit < xl) return; // Hit point x is left of leftmost edge
246  Slab::Edge const iu(slabEdges[u]);
247  Slab::EdgeXY const eXYu(slabEdgesXY[u]);
248  Vertex2D const &vu(s2d.vertices[iu]);
249  Surface2D::Edge const eu(s2d.edges[iu]);
250  Real64 const xu(vu.x + (yHit - vu.y) * eXYu);
251  if (xu < xHit) return; // Hit point is right of rightmost edge
252  while (u - l > 1u) {
253  size_type const m((l + u) / 2);
254  Slab::Edge const im(slabEdges[m]);
255  Slab::EdgeXY const eXYm(slabEdgesXY[m]);
256  Vertex2D const &vm(s2d.vertices[im]);
257  Surface2D::Edge const em(s2d.edges[im]);
258  Real64 xm(vm.x + (yHit - vm.y) * eXYm);
259  if (xHit <= xm) {
260  u = m;
261  } else {
262  l = m;
263  }
264  }
265  assert(u - l == 1u);
266  if (u % 2 == 0u) return; // Outside of nonconvex surface polygon
267  }
268  hit = true;
269 }

271 ALWAYS_INLINE
272 void PierceSurface_polygon(DataSurfaces::SurfaceData const &surface, // Surface
273  Vector3<Real64> const &hitPt, // Ray-plane intersection point
274  bool &hit // Ray intersects surface?
275 )
276 {
277  // Purpose: Check if hit point on surface plane is in surface polygon
278  //
279  // Author: Stuart Mentzer (Stuart_Mentzer@objexx.com)
280  //
281  // History:
282  // Jan 2016: Initial release
283 
284  using DataSurfaces::nVerticesBig;
285  using DataSurfaces::Surface2D;
286  using Vertex2D = Vector2<Real64>;
287  Surface2D const &s2d(surface.surface2d);
288  int const axis(s2d.axis);
289  Vertex2D const h2d(axis == 0 ? hitPt.y : hitPt.x, axis == 2 ? hitPt.y : hitPt.z); // Hit point in 2D surface's plane
290  if ((h2d.x < s2d.vl.x) || (s2d.vu.x < h2d.x) || (h2d.y < s2d.vl.y) || (s2d.vu.y < h2d.y)) return; // Misses 2D surface bounding box
291  ShapeCat const shapeCat(surface.shapeCat);
292  if (shapeCat == ShapeCat::Rectangular) { // Rectangular is most common: Special case algorithm is faster but assumes these are really rectangular
293  Vertex2D const v0h(h2d - s2d.vertices[0]);
294  Real64 const he1(v0h.dot(s2d.edges[0]));
295  if ((he1 < 0.0) || (he1 > s2d.s1)) return;
296  Real64 const he3(-v0h.dot(s2d.edges[3]));
297  if ((he3 < 0.0) || (he3 > s2d.s3)) return;
298  hit = true;
299  } else if (shapeCat == ShapeCat::Triangular) { // Cross products all nonnegative <=> Hit point in triangle
300  PierceSurface_Triangular(s2d, h2d, hit);
301  } else if ((shapeCat == ShapeCat::Nonconvex) ||
302  (s2d.vertices.size() >= nVerticesBig)) { // O( log n ) algorithm for nonconvex and many-vertex convex surfaces
303  PierceSurface_Nonconvex(s2d, h2d, hit);
304  } else if (shapeCat == ShapeCat::Convex) { // O( n ) algorithm for convex surface without too many vertices
305  PierceSurface_Convex(s2d, h2d, hit);
306  }
307 }

309 ALWAYS_INLINE
310 void PierceSurface(DataSurfaces::SurfaceData const &surface, // Surface
311  Vector3<Real64> const &rayOri, // Ray origin point
312  Vector3<Real64> const &rayDir, // Ray direction vector
313  Vector3<Real64> &hitPt, // Ray-plane intersection point
314  bool &hit // Ray intersects surface?
315 )
316 {
317  // Purpose: Check if a ray hits a surface and return the point of intersection
318  // with the surface's plane if they intersect.
319  // Convex and concave surfaces with 3 or more vertices are supported.
320  //
321  // Author: Stuart Mentzer (Stuart_Mentzer@objexx.com)
322  //
323  // History:
324  // Jan 2016: Initial release
325 
326  // Find ray intersection with surface plane
327  hit = false;
328  DataSurfaces::SurfaceData::Plane const &plane(surface.plane);
329  Real64 const den((plane.x * rayDir.x) + (plane.y * rayDir.y) + (plane.z * rayDir.z));
330  if (den == 0.0) { // Ray is parallel to plane: This not treated as piercing even if ray lies in plane
331  return;
332  } else { // Ray's line intersects plane
333  Real64 const num(-((plane.x * rayOri.x) + (plane.y * rayOri.y) + (plane.z * rayOri.z) + plane.w));
334  if (num * den <=
335  0.0) { // Ray points away from surface or ray origin is on surface: This looks odd but is fast way to check for different signs
336  return;
337  } else { // Ray points toward surface: Compute hit point
338  Real64 const t(num / den); // Ray parameter at plane intersection: hitPt = rayOri + t * rayDir
339  hitPt.x = rayOri.x + (t * rayDir.x); // Compute by coordinate to avoid Vertex temporaries
340  hitPt.y = rayOri.y + (t * rayDir.y);
341  hitPt.z = rayOri.z + (t * rayDir.z);
342  }
343  }
344 
345  // Check if hit point is in surface polygon
346  PierceSurface_polygon(surface, hitPt, hit);
347 }

371  Real64 const dMax, // Max distance from rayOri to hit point
372  Vector3<Real64> &hitPt, // Ray-plane intersection point
373  bool &hit // Ray intersects surface?
374 )
375 {
376  // Purpose: Check if a ray hits a surface and return the point of intersection
377  // with the surface's plane if they intersect.
378  // Convex and concave surfaces with 3 or more vertices are supported.
379  // This overload limits the ray-surface distance for a hit.
380  //
381  // Author: Stuart Mentzer (Stuart_Mentzer@objexx.com)
382  //
383  // History:
384  // Jan 2016: Initial release
385 
386  // Input checks
387  assert(std::abs(rayDir.mag_squared() - 1.0) < 4 * std::numeric_limits<Real64>::epsilon()); // Check unit vector
388  assert(dMax >= 0.0); // Distance must be nonnegative
389 
390  // Find ray intersection with surface plane
391  hit = false;
392  DataSurfaces::SurfaceData::Plane const &plane(surface.plane);
393  Real64 const den((plane.x * rayDir.x) + (plane.y * rayDir.y) + (plane.z * rayDir.z));
394  if (den == 0.0) { // Ray is parallel to plane: This not treated as piercing even if ray lies in plane
395  return;
396  } else { // Ray's line intersects plane
397  Real64 const num(-((plane.x * rayOri.x) + (plane.y * rayOri.y) + (plane.z * rayOri.z) + plane.w));
398  if (num * den <=
399  0.0) { // Ray points away from surface or ray origin is on surface: This looks odd but is fast way to check for different signs
400  return;
401  } else { // Ray points toward surface: Compute hit point
402  Real64 const t(num / den); // Ray parameter at plane intersection: hitPt = rayOri + t * rayDir
403  if (t > dMax) return; // Hit point exceeds distance from rayOri limit
404  hitPt.x = rayOri.x + (t * rayDir.x); // Compute by coordinate to avoid Vertex temporaries
405  hitPt.y = rayOri.y + (t * rayDir.y);
406  hitPt.z = rayOri.z + (t * rayDir.z);
407  }
408  }
409 
410  // Check if hit point is in surface polygon
411  PierceSurface_polygon(surface, hitPt, hit);
412 }
