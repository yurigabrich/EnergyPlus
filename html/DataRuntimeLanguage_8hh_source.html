488  // EMS Actuator fast duplicate check lookup support
489  typedef std::tuple<std::string, std::string, std::string> EMSActuatorKey;
490  struct EMSActuatorKey_hash : public std::unary_function<EMSActuatorKey, std::size_t>
491  {
492  inline static void hash_combine(std::size_t &seed, std::string const &s)
493  {
494  std::hash<std::string> hasher;
495  seed ^= hasher(s) + 0x9e3779b9 + (seed << 6) + (seed >> 2);
496  }
497 
498  inline std::size_t operator()(EMSActuatorKey const &key) const
499  {
500  std::size_t seed(0);
501  hash_combine(seed, std::get<0>(key));
502  hash_combine(seed, std::get<1>(key));
503  hash_combine(seed, std::get<2>(key));
504  return seed;
505  }
506  };
